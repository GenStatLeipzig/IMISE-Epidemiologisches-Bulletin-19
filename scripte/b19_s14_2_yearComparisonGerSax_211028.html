                                                                area = Bundesland, 
                                                                cases_its = Aktuelle_COVID_Faelle_ITS)]

# clean saxony data
dat_icu_sax <- dat_icu_sax[Bundesland == "SACHSEN", .(date = Datum, 
                                                      area = Bundesland, 
                                                      cases_its = Aktuelle_COVID_Faelle_ITS)]


# remove time from date entry
dat_icu_ger$date <- format(dat_icu_ger$date, format='%Y/%m/%d')

# reformate area entry

  )
